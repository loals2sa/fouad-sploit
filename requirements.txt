        function selectService(serviceId) {
            // Remove active class from all cards
            document.querySelectorAll('.service-card').forEach(card => {
                card.classList.remove('active');
            });
            
            // Add active class to selected card
            const selectedCard = document.querySelector(`.service-card[data-id="${serviceId}"]`);
            if (selectedCard) {
                selectedCard.classList.add('active');
            }
            
            // Set selected service
            selectedService = products.find(p => p.id === serviceId);
            if (selectedService) {
                document.getElementById('selectedService').value = selectedService.name;
                document.getElementById('orderBtn').disabled = false;
                calculatePrice();
                
                // Scroll to order form
                document.querySelector('.card:nth-child(2)').scrollIntoView({ behavior: 'smooth' });
            }
        }
        
        function filterServices() {
            const searchTerm = document.getElementById('serviceSearch').value.toLowerCase();
            const priceFilter = document.getElementById('priceFilter').value;
            const activeCategory = document.querySelector('.category-tab.active').dataset.category;
            
            const filtered = products.filter(service => {
                // Category filter
                if (activeCategory !== 'all' && service.category !== activeCategory) {
                    return false;
                }
                
                // Search filter
                if (searchTerm && !service.name.toLowerCase().includes(searchTerm) && 
                    !service.desc.toLowerCase().includes(searchTerm)) {
                    return false;
                }
                
                // Price filter                if (priceFilter === 'low' && service.price >= 1) return false;
                if (priceFilter === 'medium' && (service.price < 1 || service.price > 5)) return false;
                if (priceFilter === 'high' && service.price <= 5) return false;
                
                return true;
            });
            
            renderServices(filtered);
        }

            document.getElementById('serviceSearch').value = '';
            document.getElementById('priceFilter').value = 'all';
            document.querySelectorAll('.category-tab').forEach(tab => tab.classList.remove('active'));
            document.querySelector('.category-tab[data-category="all"]').classList.add('active');
            renderServices();
        }

        function calculatePrice() {
            if (!selectedService) return;
            
            const qtyInput = document.getElementById('orderQuantity');
            const errorDiv = document.getElementById('priceError');
            const breakdownDiv = document.getElementById('priceBreakdown');
            const orderBtn = document.getElementById('orderBtn');
            
            const qty = parseInt(qtyInput.value) || 0;
            errorDiv.textContent = '';
            
            // Validate quantity
            if (qty < selectedService.min) {
                errorDiv.textContent = `⚠️ أقل كمية مسموحة هي ${selectedService.min.toLocaleString()}`;
                breakdownDiv.style.display = 'none';
                orderBtn.disabled = true;
                return;
            }
            
            if (qty > selectedService.max) {
                errorDiv.textContent = `⚠️ أقصى كمية مسموحة هي ${selectedService.max.toLocaleString()}`;
                breakdownDiv.style.display = 'none';
                orderBtn.disabled = true;
                return;
            }
            
            // Calculate price
            const basePricePerUnit = selectedService.price / 1000;
            const priceWithMarkup = basePricePerUnit + (CONFIG.markup / 1000);            const total = qty * priceWithMarkup;
            
            // Update breakdown
            document.getElementById('breakdownBase').textContent = `$${basePricePerUnit.toFixed(4)}`;
            document.getElementById('breakdownMarkup').textContent = `+$${(CONFIG.markup/1000).toFixed(4)}`;
            document.getElementById('breakdownUnit').textContent = `$${priceWithMarkup.toFixed(4)}`;
            document.getElementById('breakdownQty').textContent = qty.toLocaleString();
            document.getElementById('breakdownTotal').textContent = `$${total.toFixed(4)}`;
            
            breakdownDiv.style.display = 'block';
            currentOrderTotal = total;
            orderBtn.disabled = false;
        }

        function loadOrders() {
            const list = document.getElementById('ordersList');
            
            if (orders.length === 0) {
                list.innerHTML = `
                    <div class="empty-state">
                        <i class="fas fa-clipboard-list"></i>
                        <p>لا توجد طلبات سابقة</p>
                        <button class="btn btn-outline" style="margin-top: 15px; width: auto;" onclick="document.querySelector('.service-card').click()">
                            <i class="fas fa-plus"></i> ابدأ بطلب خدمة
                        </button>
                    </div>
                `;
                return;
            }
            
            // Show only last 10 orders for performance
            const recentOrders = orders.slice(0, 10);
            
            list.innerHTML = recentOrders.map(order => {
                let statusText = 'قيد المراجعة';
                let statusClass = 'pending';
                let actionText = '';
                
                if (order.status === 'completed') {
                    statusText = 'مكتمل';
                    statusClass = 'completed';
                    actionText = order.type === 'deposit' ? '✅ تمت إضافة الرصيد' : '✅ تم تنفيذ الخدمة';
                } else if (order.status === 'rejected') {
                    statusText = 'مرفوض';
                    statusClass = 'rejected';
                    actionText = order.admin_note || '❌ لم يصل المبلغ أو بيانات غير صحيحة';
                }
                
                const isDeposit = order.type === 'deposit';
                const title = isDeposit ? 'شحن رصيد' : order.service_name;
                const amount = isDeposit ? order.amount : `$${order.amount}`;                
                return `
                <div class="order-item ${order.status} ${orders.indexOf(order) === 0 ? 'new-order' : ''}">
                    <div class="order-header">
                        <div class="order-id">${isDeposit ? '🔋' : '🛒'} ${title}</div>
                        <span class="order-status status-${statusClass}">${statusText}</span>
                    </div>
                    
                    <div class="order-details">
                        <div class="order-detail-row">
                            <span>رقم الطلب:</span>
                            <strong>${order.id}</strong>
                        </div>
                        <div class="order-detail-row">
                            <span>المبلغ:</span>
                            <strong style="color: var(--${order.status === 'completed' ? 'success' : (order.status === 'rejected' ? 'danger' : 'warning')});">
                                ${amount}
                            </strong>
                        </div>
                        <div class="order-detail-row">
                            <span>تاريخ الطلب:</span>
                            <span>${new Date(order.date).toLocaleDateString('ar-EG')} ${new Date(order.date).toLocaleTimeString('ar-EG', {hour: '2-digit', minute:'2-digit'})}</span>
                        </div>
                        ${!isDeposit ? `
                        <div class="order-detail-row">
                            <span>الرابط:</span>
                            <a href="${order.link}" target="_blank" style="color: var(--info); text-decoration: none; word-break: break-all;">
                                ${order.link.length > 35 ? order.link.substring(0, 35) + '...' : order.link}
                            </a>
                        </div>
                        <div class="order-detail-row">
                            <span>الكمية:</span>
                            <span>${order.quantity?.toLocaleString() || 'N/A'}</span>
                        </div>
                        ` : ''}
                        <div class="order-detail-row">
                            <span>رقم العملية:</span>
                            <span class="order-tx">${order.tx_id}</span>
                        </div>
                        ${actionText ? `<div class="admin-note">${actionText}</div>` : ''}
                    </div>
                </div>
                `;
            }).join('');
        }

                    if (action === 'approve') {
                        pendingOrder.status = 'completed';
                        pendingOrder.admin_note = 'تم التأكد من وصول المبلغ عبر البوت';
                        
                        // If it's a deposit, add to user balance
                        if (pendingOrder.type === 'deposit') {
                            currentUser.balance += parseFloat(pendingOrder.amount);
                            saveToStorage();
                            document.getElementById('userBalance').textContent = `$${currentUser.balance.toFixed(2)}`;
                            showToast(`✅ تمت إضافة $${pendingOrder.amount} إلى رصيدك`, 'success');
                        } else {
                            showToast('✅ تمت الموافقة على طلبك وسيتم تنفيذ الخدمة قريباً', 'success');
                        }
                    } else {
                        pendingOrder.status = 'rejected';
                        pendingOrder.admin_note = 'لم يصل المبلغ أو بيانات غير صحيحة';
                        showToast('❌ تم رفض الطلب. يرجى التأكد من بيانات التحويل', 'error');
                    }
                    
                    saveToStorage();
                    loadOrders();                    updateStats();
                    
                    // Update response
                    responseEl.innerHTML = action === 'approve' ? 
                        '<i class="fas fa-check-circle"></i> تم قبول الطلب وتحديث حالته بنجاح!' : 
                        '<i class="fas fa-times-circle"></i> تم رفض الطلب وتحديث حالته';
                    responseEl.style.background = action === 'approve' ? '#d4edda' : '#f8d7da';
                    responseEl.style.color = action === 'approve' ? '#155724' : '#721c24';
                } else {
                    responseEl.innerHTML = '<i class="fas fa-info-circle"></i> لا توجد طلبات قيد الانتظار لمحاكاتها';
                    responseEl.style.background = '#d1ecf1';
                    responseEl.style.color = '#0c5460';
                }
                
                // Re-enable buttons after delay
                setTimeout(() => {
                    btns.forEach(btn => btn.disabled = false);
                }, 2000);
                
            }, 1500);
        }

        function showToast(message, type = 'info') {
            const toast = document.getElementById('toast');
            const toastMessage = document.getElementById('toastMessage');
            const icon = toast.querySelector('i');
            
            // Set content
            toastMessage.textContent = message;
            
            // Set type
            toast.className = 'toast';
            if (type === 'success') {
                toast.classList.add('success', 'show');
                icon.className = 'fas fa-check-circle';
            } else if (type === 'error') {
                toast.classList.add('error', 'show');
                icon.className = 'fas fa-exclamation-circle';
            } else {
                toast.classList.add('show');
                icon.className = 'fas fa-info-circle';
            }
            
            // Auto hide after 5 seconds (already set in interval)
        }
